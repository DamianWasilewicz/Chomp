404    * @param foods The list of all foods from the database.

410   private void updatePriorityTable(int userID, List<Food> foods, int table, int start, int end) {
411     String statement = "UPDATE priorities" + table + " SET";
412     for (int i = start; i < end; i++) {
413       statement += " '" + i + "' = ?,";
414     }
415     statement += " '" + end + "' = ?";
416     statement += "WHERE id = ?";
417     try {
418       System.out.println(statement);
419       PreparedStatement prep = conn2.prepareStatement(statement);
420       int index = 1;
421       for (int j = start; j < end; j++) {
422         Food current = foods.get(j);
423         double currentPriority = current.getPriority();
424         prep.setDouble(index, currentPriority);
425         index++;
426       }
427       prep.setInt(index, userID);
428       prep.executeUpdate();
429       prep.close();
430     } catch (Exception e) {
431       System.out.println("Exception occured while updating priority table");
432       e.printStackTrace();
433       return;
434     }
435   }
